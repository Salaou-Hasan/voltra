// delete_user — permanently delete a user (admin or self).
// args: [username]
const [username] = args;
if (__neondb_caller_id !== username && __neondb_caller_role !== "admin") {
  result = { error: "forbidden" };
} else if (!__neondb_get("users", username)) {
  result = { error: "no_such_user" };
} else {
  __neondb_delete("users", username);
  __neondb_delete("sessions", username);
  result = { ok: true };
}
