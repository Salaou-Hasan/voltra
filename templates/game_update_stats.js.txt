// update_stats — admin-only stat override.
// args: [player_id, stats_object]
const [player_id, stats] = args;
if (__neondb_caller_role !== "admin") { result = { error: "forbidden" }; }
else {
  const p = __neondb_get("players", player_id);
  if (!p) { result = { error: "not_found" }; }
  else {
    const merged = Object.assign(p, stats || {});
    __neondb_set("players", player_id, merged);
    result = { ok: true, player: merged };
  }
}
