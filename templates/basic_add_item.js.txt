// add_item — append an item to a user's inventory.
// args: [username, item_id, qty]
const [username, item_id, qty] = args;
const key = username + ":" + item_id;
const existing = __neondb_get("inventory", key);
const next_qty = (existing && existing.qty ? existing.qty : 0) + (qty || 1);
__neondb_set("inventory", key, { username, item_id, qty: next_qty });
result = { ok: true, item_id, qty: next_qty };
