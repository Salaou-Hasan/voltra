using System;
using System.Collections.Generic;
using System.Threading.Tasks;
using UnityEngine;
using Newtonsoft.Json.Linq;

/// <summary>
/// High-level Voltra game API. Add this to a persistent GameManager GameObject.
/// Requires VoltraBehaviour on the same GameObject.
/// </summary>
public class VoltraManager : MonoBehaviour
{
    public static VoltraManager Instance { get; private set; }

    [Header("Connection")]
    public string serverUrl = "ws://localhost:3000";
    public string apiKey    = "";

    [Header("Lobby")]
    public string lobbyId = "lobby_1";

    [SerializeField] private VoltraBehaviour _db;

    /// <summary>Fired on the main thread whenever a subscribed player row changes.</summary>
    public event Action<JObject> OnPlayerUpdate;

    // -------------------------------------------------------------------------

    private void Awake()
    {
        if (Instance != null && Instance != this) { Destroy(gameObject); return; }
        Instance = this;
        DontDestroyOnLoad(gameObject);
    }

    private async void Start()
    {
        // Connect to the server
        await _db.ConnectAsync(serverUrl, apiKey);

        // Subscribe to all players in this lobby — updates arrive on the main thread
        _db.Subscribe(
            $"players WHERE lobby = '{lobbyId}'",
            row => OnPlayerUpdate?.Invoke(row)
        );
    }

    private void OnDestroy()
    {
        _db.Disconnect();
    }

    // -------------------------------------------------------------------------
    // Public game API
    // -------------------------------------------------------------------------

    /// <summary>Spawn a player into the current lobby.</summary>
    public async Task<JObject> SpawnPlayer(string playerId, string classname)
    {
        return await _db.CallAsync("spawn", new object[] { playerId, lobbyId, classname });
    }

    /// <summary>Update a player's position.</summary>
    public async Task<JObject> MovePlayer(string playerId, float x, float y)
    {
        return await _db.CallAsync("move", new object[] { playerId, x, y });
    }

    /// <summary>Remove a player from the world.</summary>
    public async Task<JObject> DespawnPlayer(string playerId)
    {
        return await _db.CallAsync("despawn", new object[] { playerId });
    }

    /// <summary>
    /// Example: wire to Unity's Update loop to flush main-thread callbacks.
    /// VoltraBehaviour.Update() handles this automatically when attached to the
    /// same GameObject. No manual call needed.
    /// </summary>
    private void Update()
    {
        // VoltraBehaviour pumps its MainThreadQueue in its own Update().
        // Put any per-frame game logic here (interpolation, camera, etc.)
    }
}
