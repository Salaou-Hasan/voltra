// hello — increment a counter.
// args: [name, delta]
const [name, delta] = args;
const current = __neondb_get("counters", name);
const value = (current && typeof current.value === "number" ? current.value : 0) + (delta || 1);
__neondb_set("counters", name, { name, value });
result = { name, value };
