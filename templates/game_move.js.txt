// ═══════════════════════════════════════════════════════════════════════
// move — update a player's position and zone
// ───────────────────────────────────────────────────────────────────────
// Args:   [player_id: string, x: number, y: number]
// Return: { ok: true, x: number, y: number, zone: string }
//         { error: "not_found" }
//
// Usage:
//   neondb call move '["player1", 150, 75]'
//
// NeonDB reducer API:
//   __neondb_get(table, key)      → object | null
//   __neondb_set(table, key, val) → void
//   __neondb_delete(table, key)   → void
//   __neondb_get_all(table)       → object[]
//   __neondb_caller_id            → string (client identity)
//   __neondb_caller_role          → string (e.g. "admin")
//
// Performance: runs on Boa JS engine (no JIT).
//   Run `neondb build` to compile to WASM for 10–50× more throughput.
// ═══════════════════════════════════════════════════════════════════════
function reducer(args) {
  const [player_id, x, y] = args;
  const p = __neondb_get("players", player_id);
  if (!p) { return { error: "not_found" }; }
  else {
    p.x = x; p.y = y;
    p.zone = "zone_" + Math.floor(x / 100) + "_" + Math.floor(y / 100);
    __neondb_set("players", player_id, p);
    return { ok: true, x, y, zone: p.zone };
  }
}
