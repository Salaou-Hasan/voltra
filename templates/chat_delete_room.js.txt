// delete_room — owner-only room deletion.
// args: [room_id]
const [rid] = args;
const r = __neondb_get("rooms", rid);
if (!r) { result = { error: "no_such_room" }; }
else if (r.owner !== __neondb_caller_id && __neondb_caller_role !== "admin") {
  result = { error: "forbidden" };
} else {
  __neondb_delete("rooms", rid);
  result = { ok: true };
}
