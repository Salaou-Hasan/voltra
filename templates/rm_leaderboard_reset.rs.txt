use neondb::{reducer, ret};

#[reducer]
fn reset_leaderboard(ctx: Ctx, board: String) {
    let prefix = format!("{}:", board);
    let rows = ctx.tables.list_rows_with_keys("leaderboard");
    let mut count = 0i64;

    for (key, _) in rows {
        if key.starts_with(&prefix) {
            ctx.delete("leaderboard", &key)?;
            count += 1;
        }
    }

    ret!({ "ok": true, "count": count })
}
